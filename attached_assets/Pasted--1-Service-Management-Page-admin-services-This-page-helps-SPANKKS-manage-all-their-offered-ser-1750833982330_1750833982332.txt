const csv = require('csv-parser');
const fs = require('fs');

function parseCSVUpload(filePath, onFinish) {
  const results = [];
  fs.createReadStream(filePath)
    .pipe(csv())
    .on('data', (data) => results.push(data))
    .on('end', () => onFinish(results));
}
